def load_graphics(path,accept=(.jpg','.png','.bap','.gif')): # jia zai tu pian de han shu, er zhe ge hanshu zhi jieshou liangge can shu, path = cun fang tu pian de wen jian jia , accecpt= jie shou de wen jian 
	graphics={}
	for pic in os.path.splictext(path)  #path.slictext= bian li wen jian jia 
	name,ext= os.path.splixtext(pic) # chai fen wen jian , wen jian ming + hou zhui( extention)
	if ext.lower() in accpt:
		img=pygame.image.load(os.path.join(path.pic))# zai ru tu pian 
		if img.get_alpha(): #alpha ceng = tou ming di 
			img=img.convert_alpha() #  gai cheng tou ming ceng de ge shi 
		else:
			img=img.convert()
		graphics[name]=img
	return graphics

def get_image(sheet,x,y,width,height,colorkey,scale):# sheet chuan ru yi zhang tu pian 
	image=pygame.Surface((width,height))
	image.blit(sheet,(0,0),(x,y,width,height)) # 0,0 dai biao hua dao na li, x,y,w,h dai biao
